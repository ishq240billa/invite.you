        // Access Controller Logic
        function verifyAndProceed() {
            const parsedName = document.getElementById('nameVerification').value.trim();
            if(!parsedName) {
                alert("Access Denied: Formal identification requires a name.");
                return;
            }
            
            document.getElementById('insertedName').innerText = parsedName;
            document.getElementById('portalStage').classList.add('hidden');
            document.getElementById('exhibitionStage').classList.remove('hidden');
        }

        // --- HIGH VELOCITY PARTY POPPER SIMULATION ---
        const canvas = document.getElementById('particleCanvas');
        const ctx = canvas.getContext('2d');
        let particleArray = [];

        function setViewport() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', setViewport);
        setViewport();

        class ExecutiveParticle {
            constructor(originX, originY) {
                this.x = originX;
                this.y = originY;
                const palette = ['#bf953f', '#b38728', '#111e16', '#de4437', '#2a754d', '#4285f4', '#ffffff'];
                this.color = palette[Math.floor(Math.random() * palette.length)];
                this.dimensions = Math.random() * 8 + 4;
                
                // Pure 360-degree radical directional velocity computation
                const trajectoryAngle = Math.random() * Math.PI * 2;
                const force = Math.random() * 15 + 8;
                this.vx = Math.cos(trajectoryAngle) * force;
                this.vy = Math.sin(trajectoryAngle) * force;
                this.downwardPull = 0.28;
                this.alpha = 1;
                this.fadingRate = Math.random() * 0.012 + 0.008;
            }
            process() {
                this.vy += this.downwardPull;
                this.x += this.vx;
                this.y += this.vy;
                this.alpha -= this.fadingRate;
            }
            render() {
                ctx.save();
                ctx.globalAlpha = this.alpha;
                ctx.fillStyle = this.color;
                ctx.fillRect(this.x, this.y, this.dimensions, this.dimensions);
                ctx.restore();
            }
        }

        function generateConcentricBurst(x, y) {
            for(let i=0; i<180; i++) {
                particleArray.push(new ExecutiveParticle(x, y));
            }
        }

        function executionLoop() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            particleArray = particleArray.filter(item => item.alpha > 0);
            particleArray.forEach(item => {
                item.process();
                item.render();
            });
            if(particleArray.length > 0) {
                requestAnimationFrame(executionLoop);
            }
        }

        // 3D Presentation Interaction Script
        function execute3DUnboxing(event) {
            event.stopPropagation();
            
            // Calculate pixel precise anchor point for the blast vectors from the physical seal center
            const boundary = event.target.getBoundingClientRect();
            const anchorX = boundary.left + boundary.width / 2;
            const anchorY = boundary.top + boundary.height / 2;

            // Trigger mechanical classes
            document.getElementById('envelopeRig').classList.add('unboxed');

            // Initialize particle burst sequence
            generateConcentricBurst(anchorX, anchorY);
            executionLoop();
        }
    </script>
